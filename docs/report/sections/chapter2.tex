\chapter{Phương Pháp Luận \& Kiến Trúc Hệ Thống}

\section{Phương pháp luận thiết kế hệ thống}
Để xây dựng thành công hệ thống vi xử lý đa lõi 8-Core RISC-V hoạt động ổn định và tối ưu, dự án áp dụng phương pháp luận thiết kế phần cứng hệ thống theo các nguyên tắc cốt lõi sau:

\subsection{Quy trình thiết kế Top-Down kết hợp Bottom-Up}
Quy trình xây dựng hệ thống kết hợp hài hòa hai cách tiếp cận:
\begin{itemize}
    \item \textbf{Thiết kế từ trên xuống (Top-Down Design):} Xuất phát từ yêu cầu hệ thống đa nhân chia sẻ bộ nhớ và sử dụng tập lệnh RISC-V. Hệ thống được phân rã thành các khối chức năng độc lập bao gồm: Lõi đơn nhân (Core), Bộ phân xử (Arbiter), Bộ điều khiển Bus (Bus Controller) và Bộ nhớ dữ liệu dùng chung (Shared RAM). Việc xác định rõ giao tiếp (Interface) giữa các khối này là bước tiên quyết trước khi viết mã RTL.
    \item \textbf{Tích hợp từ dưới lên (Bottom-Up Integration):} Hiện thực hóa và kiểm thử kỹ lưỡng từ mức linh kiện cơ bản: Đơn vị số học logic (ALU), Tập thanh ghi (Register File), Bộ điều khiển (Control Unit), Logic cập nhật PC. Sau đó tích hợp thành Lõi xử lý đơn (Core) và trục liên kết đa nhân (Interconnect). Cuối cùng là tích hợp toàn hệ thống 8-Core và kiểm chứng bằng testbench hệ thống phức tạp.
\end{itemize}

\subsection{Mô hình hóa phần cứng ở mức RTL (RTL Modeling)}
Toàn bộ hệ thống được mô tả bằng ngôn ngữ mô tả phần cứng \textbf{Verilog HDL} (chuẩn IEEE 1364-2001). Phương pháp thiết kế RTL tập trung vào:
\begin{itemize}
    \item \textbf{Tách biệt mạch tuần tự và mạch tổ hợp:} Sử dụng sườn lên xung nhịp (\texttt{posedge clk}) để chốt dữ liệu vào các thanh ghi pipeline và bộ đếm chương trình PC. Sử dụng khối \texttt{always @(*)} và \texttt{assign} cho các mạch tổ hợp giải mã lệnh và tính toán ALU.
    \item \textbf{Thiết kế modular hóa:} Mỗi khối chức năng được đóng gói trong một file riêng biệt, giao tiếp qua các cổng dữ liệu (ports) rõ ràng, giúp dễ dàng bảo trì và kiểm thử độc lập.
\end{itemize}

\subsection{Quy trình mô phỏng và kiểm chứng dựa trên công cụ mã nguồn mở}
Kiểm chứng chức năng (Functional Verification) đóng vai trò sống còn. Phương pháp luận kiểm chứng của dự án dựa trên:
\begin{itemize}
    \item \textbf{Mô phỏng mức RTL (RTL Simulation):} Sử dụng công cụ mã nguồn mở \textbf{Icarus Verilog} để biên dịch mã nguồn và chạy các testbench. Các kịch bản kiểm thử được thiết kế để bao phủ các tình huống xung đột dữ liệu (data hazard), xung đột điều khiển (control hazard) và tranh chấp bus bộ nhớ.
    \item \textbf{Phân tích giản đồ dạng sóng (Waveform Analysis):} Xuất tệp dạng sóng định dạng VCD (Value Change Dump) và sử dụng phần mềm trực quan \textbf{GTKWave} để gỡ lỗi (debug) các hành vi phần cứng theo từng chu kỳ xung nhịp.
\end{itemize}

\subsection{Mô hình hóa chia sẻ tài nguyên và cơ chế đồng bộ}
Đối với hệ thống 8 lõi dùng chung bộ nhớ:
\begin{itemize}
    \item \textbf{Tránh deadlock và starvation:} Áp dụng thuật toán phân xử Round-Robin cho bộ Arbiter nhằm cấp quyền truy cập bus tuần hoàn, công bằng giữa các nhân.
    \item \textbf{Đồng bộ hóa phần cứng:} Hỗ trợ các chỉ thị lệnh nguyên tử (Atomic) từ nhóm mở rộng A-Extension nhằm giải quyết xung đột ghi/đọc đồng thời từ phần mềm đa nhân.
\end{itemize}

\section{Kiến trúc tập lệnh RISC-V (ISA) áp dụng trong dự án}
Trước khi đi vào thiết kế phần cứng ở mức RTL, việc hiểu rõ kiến trúc tập lệnh (Instruction Set Architecture - ISA) là bắt buộc. ISA đóng vai trò là đặc tả giao tiếp giữa phần mềm và phần cứng. Trong dự án này, hệ thống vi xử lý được xây dựng dựa trên tập chỉ thị \textbf{RV32I} kết hợp với một số chỉ thị thuộc nhóm mở rộng nguyên tử \textbf{A-Extension} (RV32IA subset).

\subsection{Tổng quan về tập chỉ thị RV32I}
RV32I là tập chỉ thị số nguyên cơ bản dạng 32-bit của RISC-V. Đây là tập chỉ thị bắt buộc và là nền tảng cho mọi bộ xử lý RISC-V. Các đặc trưng kỹ thuật chính bao gồm:
\begin{itemize}
    \item \textbf{Độ rộng địa chỉ và thanh ghi:} 32-bit cho cả thanh ghi đa năng và không gian địa chỉ bộ nhớ.
    \item \textbf{Cơ chế nạp/lưu (Load-Store Architecture):} Chỉ các chỉ thị Load (\texttt{LW}) và Store (\texttt{SW}) mới được phép truy cập bộ nhớ dữ liệu. Các phép toán toán học và logic chỉ thực hiện trực tiếp trên tập thanh ghi.
    \item \textbf{Độ dài lệnh cố định:} Mọi chỉ thị lệnh đều có độ rộng cố định là 32-bit, giúp đơn giản hóa quá trình nạp và giải mã lệnh trong pipeline phần cứng.
\end{itemize}

\subsection{Các kiểu định dạng chỉ thị lệnh cơ bản}
Kiến trúc RISC-V phân chia mã lệnh 32-bit thành 6 kiểu định dạng cơ bản để tối ưu bộ giải mã:
\begin{itemize}
    \item \textbf{R-type (Register):} Dành cho các lệnh tính toán số học/logic giữa các thanh ghi nguồn (\texttt{rs1}, \texttt{rs2}) và ghi kết quả vào thanh ghi đích (\texttt{rd}).
    \item \textbf{I-type (Immediate):} Dành cho các lệnh tính toán với hằng số nạp trực tiếp 12-bit (\texttt{imm}) hoặc lệnh Load bộ nhớ.
    \item \textbf{S-type (Store):} Dành cho chỉ thị Store lưu dữ liệu từ thanh ghi nguồn vào bộ nhớ.
    \item \textbf{B-type (Branch):} Dành cho chỉ thị so sánh rẽ nhánh có điều kiện với độ lệch địa chỉ 12-bit.
    \item \textbf{U-type (Upper Immediate):} Dành cho chỉ thị nạp hằng số lớn 20-bit vào nửa cao của thanh ghi (như \texttt{LUI}, \texttt{AUIPC}).
    \item \textbf{J-type (Jump):} Dành cho chỉ thị nhảy không điều kiện (\texttt{JAL}).
\end{itemize}

\subsection{Đặc tả định dạng mã hóa và trường bit của các lệnh thực hiện}
Bộ xử lý trong dự án này hiện thực một tập con của kiến trúc tập lệnh RV32I cùng nhóm mở rộng nguyên tử A-Extension để điều khiển và đồng bộ tính toán đa lõi. Dưới đây là bảng đặc tả chi tiết mã hóa nhị phân 32-bit cho từng chỉ thị lệnh được triển khai thực tế trong mã nguồn RTL:

\subsubsection{1. Nhóm lệnh định dạng R (R-type)}
Các lệnh tính toán số học và logic thực thi trực tiếp giữa các thanh ghi nguồn (\texttt{rs1}, \texttt{rs2}) và lưu kết quả vào thanh ghi đích (\texttt{rd}). Chúng sử dụng chung mã opcode là \texttt{0110011} (7'h33). Chi tiết mã hóa các trường bit được trình bày trong Bảng \ref{tab:enc_r}:

\subsubsection{2. Nhóm lệnh định dạng I (I-type)}
Các lệnh tính toán với hằng số nạp trực tiếp (\texttt{imm}), lệnh nạp dữ liệu từ bộ nhớ (\texttt{LW}), hoặc lệnh nhảy gián tiếp (\texttt{JALR}) có độ dài hằng số là 12-bit.
\begin{itemize}
    \item Lệnh tính toán ALU sử dụng opcode \texttt{0010011} (7'h13).
    \item Lệnh đọc bộ nhớ \texttt{LW} sử dụng opcode \texttt{0000011} (7'h03).

\end{enumerate}

\subsection{Tầng Nạp chỉ thị (Instruction Fetch - IF)}
Nhiệm vụ chính của tầng IF là nạp mã lệnh 32-bit từ bộ nhớ lệnh cục bộ (\texttt{instr\_mem.v}) tại địa chỉ hiện tại của bộ đếm chương trình PC và chuẩn bị dữ liệu cho tầng giải mã tiếp theo.

\begin{itemize}
    \item \textbf{Cơ chế cập nhật và quản lý địa chỉ PC:} Bộ đếm chương trình (\texttt{pc}) được cập nhật đồng bộ ở mỗi sườn lên của xung nhịp (\texttt{posedge clk}). Địa chỉ tiếp theo được lựa chọn thông qua bộ chọn MUX điều khiển PC với mức độ ưu tiên giảm dần từ trên xuống dưới:
    \begin{enumerate}
        \item \textbf{Tín hiệu Reset:} Khi tín hiệu \texttt{reset} ở mức hoạt động cao, PC được thiết lập về giá trị khởi đầu mặc định \texttt{32'd0}.
        \item \textbf{Tín hiệu Stall (Đóng băng Pipeline):} Khi phát hiện xung đột dữ liệu Load-Use ở tầng giải mã (\texttt{load\_use\_stall = 1}) hoặc lõi đang phải đợi giao dịch bộ nhớ dùng chung hoàn tất ở tầng truy cập bộ nhớ (\texttt{mem\_wait = 1}), PC sẽ giữ nguyên giá trị cũ nhằm tạm dừng nạp lệnh mới: \texttt{pc <= pc}.
        \item \textbf{Tín hiệu chuyển hướng chương trình (Redirect/Jump):} Khi xảy ra rẽ nhánh thành công hoặc thực thi lệnh nhảy không điều kiện được xác định ở tầng EX (\texttt{redirect\_taken = 1}), PC sẽ nhận địa chỉ đích mới từ đường hồi tiếp: \texttt{pc <= redirect\_target}.
        \item \textbf{Tăng tuần tự (Default):} Trong điều kiện thực thi bình thường, địa chỉ PC tự động tăng tiến thêm 4 byte sau mỗi chu kỳ để chỉ đến lệnh tiếp theo: \texttt{pc <= pc + 32'd4}.
    \end{enumerate}
    
    \item \textbf{Bộ nhớ lệnh (\texttt{instr\_mem.v}):} Là bộ nhớ lưu trữ chương trình rộng 32-bit với dung lượng 256 từ (\texttt{PROGRAM\_WORDS = 256}). Tầng IF thực hiện truy cập bộ nhớ này tại địa chỉ PC để đọc mã lệnh tương ứng ngay trong chu kỳ hiện tại.
    
    \item \textbf{Thanh ghi đệm Pipeline IF/ID:} Tại cuối chu kỳ nạp lệnh, nếu pipeline không bị stall, mã lệnh đọc ra (\texttt{instr\_from\_mem}) cùng với địa chỉ PC tương ứng sẽ được lưu vào các thanh ghi đệm \texttt{if\_id\_instr} và \texttt{if\_id\_pc}. Khi xảy ra rẽ nhánh hoặc nhảy (\texttt{redirect\_taken = 1}), cơ chế phần cứng sẽ kích hoạt quá trình xóa (\texttt{Flush}) bằng cách xóa cờ hợp lệ \texttt{if\_id\_valid <= 1'b0} và đưa mã lệnh đệm về \texttt{32'd0} để chuyển đổi lệnh kế tiếp thành lệnh rỗng (NOP).
\end{itemize}

\subsection{Tầng Giải mã chỉ thị (Instruction Decode - ID)}
Tầng ID nhận chỉ thị lệnh 32-bit từ thanh ghi đệm \texttt{if\_id\_instr} để tiến hành giải mã song song các tín hiệu điều khiển, đọc dữ liệu từ tập thanh ghi đa năng và phát hiện các xung đột dữ liệu sớm.

\begin{itemize}
    \item \textbf{Khối giải mã trung tâm (\texttt{control\_unit.v}):} Khối này phân tách mã máy thành các trường bit chức năng và sinh ra tập hợp các tín hiệu điều khiển cấu hình phần cứng:
    \begin{itemize}
        \item \texttt{cu\_reg\_write}: Cho phép ghi kết quả vào tập thanh ghi ở tầng WB.
        \item \texttt{cu\_alu\_src}: Điều khiển bộ chọn MUX ngõ vào thứ hai của ALU ($0$: chọn thanh ghi nguồn \texttt{rs2\_data}, $1$: chọn hằng số mở rộng \texttt{cu\_imm}).
        \item \texttt{cu\_alu\_control[3:0]}: Mã hóa phép toán chi tiết cho ALU.
        \item \texttt{cu\_mem\_read} / \texttt{cu\_mem\_write}: Kích hoạt chu kỳ đọc/ghi bộ nhớ dùng chung.
        \item \texttt{cu\_mem\_to\_reg}: Chọn nguồn ghi ngược ($0$: kết quả tính toán ALU, $1$: dữ liệu đọc từ RAM).
        \item Các tín hiệu đồng bộ nguyên tử (\texttt{cu\_mem\_lr}, \texttt{cu\_mem\_sc}, \texttt{cu\_mem\_amo}, \texttt{cu\_amo\_op[3:0]}) phục vụ cho nhóm lệnh mở rộng A-Extension.
    \end{itemize}
    
    \item \textbf{Bộ tạo hằng số mở rộng (Immediate Generator):} Thực hiện trích xuất trường hằng số trong mã lệnh và tiến hành mở rộng dấu (Sign-extension) thành hằng số 32-bit \texttt{cu\_imm} phù hợp với định dạng của từng kiểu lệnh RISC-V (I, S, B, U, J).
    
    \item \textbf{Tập thanh ghi đa năng (\texttt{reg\_file.v}):} Thực hiện đọc bất đồng bộ (asynchronous read) dữ liệu từ hai thanh ghi nguồn \texttt{rs1} và \texttt{rs2} thông qua hai địa chỉ giải mã \texttt{cu\_rs1\_addr} và \texttt{cu\_rs2\_addr}. Việc đọc bất đồng bộ giúp dữ liệu toán hạng sẵn sàng ngay trong chu kỳ ID để kịp tính toán ở chu kỳ EX kế tiếp.
    
    \item \textbf{Xử lý xung đột dữ liệu bằng cơ chế Load-Use Stall:}
    Khi lệnh ở tầng Execute (EX) là lệnh đọc bộ nhớ (như load, lr, sc, amo) và thanh ghi đích của nó (\texttt{id\_ex\_rd\_addr}) trùng với một trong hai thanh ghi nguồn đang được giải mã ở tầng ID (\texttt{cu\_rs1\_addr} hoặc \texttt{cu\_rs2\_addr}), dữ liệu chưa kịp nạp về để forward. Phần cứng sẽ tự động phát hiện xung đột này thông qua biểu thức logic:
    \[
    \begin{aligned}
    \text{load\_use\_stall} = {} & \text{IF/ID.valid} \ \&\ \text{ID/EX.load\_like} \ \&\ (\text{ID/EX.rd} \neq 5'd0) \ \& \\
    & [(\text{ID/EX.rd} == \text{IF/ID.rs1}) \ \parallel\ (\text{ID/EX.rd} == \text{IF/ID.rs2})]
    \end{aligned}
    \]
    Khi \texttt{load\_use\_stall} được kích hoạt, phần cứng sẽ thực hiện đóng băng tầng IF (giữ nguyên địa chỉ PC và thanh ghi đệm \texttt{if\_id}) đồng thời bơm một chỉ thị rỗng (NOP) vào tầng EX bằng cách thiết lập cờ hợp lệ \texttt{id\_ex\_valid <= 1'b0} và vô hiệu hóa các tín hiệu điều khiển ghi thanh ghi, ghi bộ nhớ của tầng EX.
    
    \item \textbf{Thanh ghi đệm Pipeline ID/EX:} Khi không bị stall, tất cả dữ liệu toán hạng (\texttt{rs1\_data}, \texttt{rs2\_data}, \texttt{cu\_imm}), địa chỉ thanh ghi (\texttt{rs1\_addr}, \texttt{rs2\_addr}, \texttt{rd\_addr}) cùng toàn bộ các tín hiệu điều khiển giải mã sẽ được chốt vào thanh ghi đệm \texttt{id\_ex\_*} ở sườn lên clock kế tiếp.
\end{itemize}

\subsection{Tầng Thực thi (Execute - EX)}
Tầng EX là trung tâm xử lý số học, logic và điều khiển luồng chương trình. Tại đây, các phép toán được thực thi bởi ALU, đồng thời các mạch phụ trợ thực hiện bơm tắt dữ liệu và xử lý rẽ nhánh/nhảy.

\begin{itemize}
    \item \textbf{Mạch Forwarding (Bơm tắt kết quả) giải quyết xung đột dữ liệu:} 
    Để loại bỏ các chu kỳ trễ (stalls) khi các lệnh phụ thuộc dữ liệu đi liền nhau, hệ thống tích hợp khối logic forwarding cho phép lấy trực tiếp kết quả đang lưu giữ trên các tầng pipeline phía trước:
    \begin{itemize}
        \item \textbf{Forward từ tầng EX/MEM:} Điều kiện \texttt{ex\_mem\_can\_forward} kích hoạt khi lệnh ở tầng EX/MEM ghi thanh ghi (\texttt{ex\_mem\_reg\_write = 1}), địa chỉ đích khác thanh ghi mặc định (\texttt{ex\_mem\_rd\_addr != 5'd0}) và không phải là lệnh đọc bộ nhớ (như load, lr, sc, amo).
        \item \textbf{Forward từ tầng MEM/WB:} Điều kiện \texttt{mem\_wb\_can\_forward} kích hoạt khi lệnh ở tầng MEM/WB ghi thanh ghi (\texttt{mem\_wb\_reg\_write = 1}) và địa chỉ đích khác thanh ghi mặc định (\texttt{mem\_wb\_rd\_addr != 5'd0}).
        \item \textbf{Biểu thức chọn dữ liệu toán hạng:}
        \[
        \text{fwd\_rs1\_data} = 
        \begin{cases} 
        \text{ex\_mem\_result} & \text{nếu } \text{EX\_MEM.rd} == \text{ID\_EX.rs1} \\
        \text{mem\_wb\_write\_data} & \text{nếu } \text{MEM\_WB.rd} == \text{ID\_EX.rs1} \\
        \text{id\_ex\_rs1\_data} & \text{mặc định}
        \end{cases}
        \]
        (Logic tương tự được áp dụng để tính \texttt{fwd\_rs2\_data} cho toán hạng nguồn thứ hai).
    \end{itemize}
    
    \item \textbf{Bộ chọn nguồn vào ALU (ALU Source MUX):} 
    Toán hạng thứ nhất \texttt{alu\_input\_a} luôn được kết nối với \texttt{fwd\_rs1\_data}. Toán hạng thứ hai \texttt{alu\_input\_b} được chọn qua một bộ MUX 2-sang-1 dựa trên tín hiệu chọn nguồn \texttt{id\_ex\_alu\_src}: chọn hằng số mở rộng \texttt{id\_ex\_imm} nếu \texttt{id\_ex\_alu\_src = 1}, ngược lại chọn \texttt{fwd\_rs2\_data} nếu \texttt{id\_ex\_alu\_src = 0}.
    
    \item \textbf{Khối số học logic ALU (\texttt{alu.v}):} 
    ALU tính toán kết quả \texttt{alu\_result} và cờ trạng thái \texttt{alu\_zero} dựa trên mã điều khiển phép toán \texttt{id\_ex\_alu\_control[3:0]}.
    
    \item \textbf{Logic tính toán kết quả của tầng Execute (\texttt{ex\_stage\_result}):}
    Tùy thuộc vào lệnh đang thực thi, kết quả đầu ra của tầng EX có thể khác nhau:
    \begin{itemize}
        \item Lệnh \texttt{LUI}: \texttt{ex\_stage\_result = id\_ex\_imm}.
        \item Lệnh \texttt{AUIPC}: \texttt{ex\_stage\_result = id\_ex\_pc + id\_ex\_imm}.
        \item Lệnh nhảy \texttt{JAL} hoặc \texttt{JALR}: \texttt{ex\_stage\_result = id\_ex\_pc + 32'd4} (lưu địa chỉ trả về của hàm).
        \item Các lệnh khác: \texttt{ex\_stage\_result = alu\_result}.
    \end{itemize}
    
    \item \textbf{Logic điều khiển rẽ nhánh và nhảy (Giải quyết xung đột điều khiển):} 
    Phần cứng đánh giá điều kiện rẽ nhánh (BEQ, BNE, BLT, BGE) dựa trên các giá trị toán hạng đã forward (\texttt{fwd\_rs1\_data} và \texttt{fwd\_rs2\_data}). Khi phát hiện điều kiện rẽ nhánh thỏa mãn hoặc lệnh là lệnh nhảy (\texttt{id\_ex\_jump = 1}):
    \begin{itemize}
        \item Tín hiệu chuyển hướng \texttt{redirect\_taken} được kéo lên mức cao ($1$).
        \item Địa chỉ đích được tính toán: đối với \texttt{JALR} là \texttt{(fwd\_rs1\_data + id\_ex\_imm) \& 32'hFFFF\_FFFE}; đối với các lệnh rẽ nhánh hoặc \texttt{JAL} là \texttt{id\_ex\_pc + id\_ex\_imm}.
        \item Tín hiệu \texttt{redirect\_taken = 1} được gửi ngược về tầng IF để cập nhật PC, đồng thời thực hiện \textbf{Flush} toàn bộ các lệnh đang nằm trong tầng IF/ID và ID/EX để ngăn chặn việc thực thi các chỉ thị sai luồng chương trình.
    \end{itemize}
\end{itemize}

\subsection{Tầng Truy cập bộ nhớ (Memory Access - MEM)}
Tầng MEM chịu trách nhiệm thực hiện các giao dịch đọc và ghi dữ liệu với bộ nhớ ngoài thông qua giao tiếp bus của lõi CPU.

\begin{itemize}
    \item \textbf{Định tuyến yêu cầu truy cập bộ nhớ:} 
    Khi lệnh ở tầng EX/MEM hợp lệ (\texttt{ex\_mem\_valid = 1}) và yêu cầu truy cập bộ nhớ (như lệnh Load, Store hoặc lệnh nguyên tử), phần cứng sẽ sinh tín hiệu yêu cầu \texttt{mem\_req = 1} và xuất dữ liệu cùng các tín hiệu điều khiển ra cổng giao tiếp bộ nhớ:
    \begin{itemize}
        \item Địa chỉ bộ nhớ: \texttt{mem\_addr = ex\_mem\_result}.
        \item Dữ liệu ghi: \texttt{mem\_wdata = ex\_mem\_store\_data}.
        \item Tín hiệu cho phép ghi: \texttt{mem\_we = ex\_mem\_mem\_write || ex\_mem\_mem\_sc || ex\_mem\_mem\_amo}.
        \item Tín hiệu cho phép đọc: \texttt{mem\_re = ex\_mem\_mem\_read || ex\_mem\_mem\_lr || ex\_mem\_mem\_amo}.
        \item Các tín hiệu điều khiển nguyên tử: \texttt{mem\_lr}, \texttt{mem\_sc}, \texttt{mem\_amo}, và mã phép toán \texttt{mem\_amo\_op[3:0]}.
    \end{itemize}
    
    \item \textbf{Đồng bộ hóa đợi bộ nhớ (Cơ chế Memory Stall):} 
    Do giao dịch truy cập bộ nhớ có thể kéo dài nhiều chu kỳ xung nhịp tùy thuộc vào phản hồi từ bộ nhớ ngoài, lõi CPU cần cơ chế đồng bộ hóa: nếu lõi phát yêu cầu nhưng tín hiệu sẵn sàng từ bên ngoài chưa được khẳng định (\texttt{mem\_ready = 0}):
    \begin{itemize}
        \item Tín hiệu hoãn pipeline \texttt{mem\_wait} được kích hoạt lên mức cao ($1$).
        \item Tín hiệu này sẽ tạm thời đóng băng (freeze) việc cập nhật của tất cả các thanh ghi đệm pipeline ở các tầng trước (IF, ID, EX) nhằm bảo toàn trạng thái và dữ liệu cho đến khi giao dịch bộ nhớ hoàn thành.
    \end{itemize}
    
    \item \textbf{Chốt dữ liệu phản hồi và hoàn thành chu kỳ:} 
    Khi nhận được tín hiệu bộ nhớ sẵn sàng (\texttt{mem\_ready = 1}):
    \begin{itemize}
        \item Giao dịch hoàn tất, dữ liệu phản hồi được chốt vào thanh ghi đệm \texttt{mem\_wb\_write\_data}: nếu là lệnh đọc bộ nhớ (\texttt{ex\_mem\_mem\_read/lr/amo}), dữ liệu nhận về là \texttt{mem\_rdata}; nếu là lệnh ghi điều kiện (\texttt{ex\_mem\_mem\_sc}), dữ liệu nhận về là cờ trạng thái \texttt{\{31'b0, mem\_sc\_result\}} (thành công trả về 0, thất bại trả về 1); đối với các lệnh khác không truy cập bộ nhớ, giá trị \texttt{ex\_mem\_result} được truyền thẳng qua.
        \item Thanh ghi đệm pipeline \texttt{ex\_mem\_valid} được xóa về \texttt{1'b0} trong một chu kỳ để đảm bảo cổng bus của lõi quay về trạng thái rỗi trước khi có thể thực hiện giao dịch tiếp theo.
    \end{itemize}
\end{itemize}

\subsection{Tầng Ghi ngược (Write Back - WB)}
Tầng WB là giai đoạn cuối cùng trong đường truyền dữ liệu pipeline 5 tầng. Nhiệm vụ duy nhất của giai đoạn này là ghi đồng bộ kết quả thực thi lệnh trở lại tập thanh ghi đa năng.

\begin{itemize}
    \item \textbf{Mạch chọn ghi ngược (Write Back MUX):} 
    Dữ liệu được lựa chọn để ghi ngược về thanh ghi đích (\texttt{mem\_wb\_write\_data}) được quyết định dựa trên nguồn dữ liệu tương ứng của lệnh ở tầng MEM:
    \begin{itemize}
        \item Đối với lệnh đọc bộ nhớ (như \texttt{LW}, \texttt{LR.W}, hoặc kết quả lệnh nguyên tử \texttt{AMO}): chọn dữ liệu vừa nhận được từ Bus (\texttt{mem\_rdata}).
        \item Đối với lệnh ghi điều kiện nguyên tử (\texttt{SC.W}): chọn cờ trạng thái kết quả \texttt{\{31'b0, mem\_sc\_result\}}.
        \item Đối với các lệnh số học logic, lệnh tính địa chỉ hoặc lệnh nhảy: truyền thẳng kết quả tính toán của ALU (\texttt{ex\_mem\_result}).
    \end{itemize}
    
    \item \textbf{Cập nhật đồng bộ vào Tập thanh ghi (\texttt{reg\_file.v}):} 
    Tại sườn lên của xung nhịp tiếp theo, dữ liệu từ tầng WB được ghi vào tập thanh ghi nếu thỏa mãn đồng thời các điều kiện sau:
    \begin{itemize}
        \item Chỉ thị hiện tại ở tầng WB là hợp lệ (\texttt{mem\_wb\_valid = 1}).
        \item Tín hiệu cho phép ghi thanh ghi được kích hoạt từ khối giải mã (\texttt{mem\_wb\_reg\_write = 1}).
        \item Địa chỉ thanh ghi ghi đích khác thanh ghi \texttt{x0} (\texttt{mem\_wb\_rd\_addr != 5'd0}). Mạch phần cứng bên trong \texttt{reg\_file.v} luôn nối cứng thanh ghi \texttt{x0} xuống đất (GND), do đó mọi yêu cầu ghi vào địa chỉ \texttt{5'd0} đều tự động bị bỏ qua, giữ cho giá trị của \texttt{x0} luôn bằng 0.
    \end{itemize}
    Tín hiệu cho phép ghi tập thanh ghi được định nghĩa bởi biểu thức logic kết hợp: \texttt{rf\_write\_en = mem\_wb\_valid \&\& mem\_wb\_reg\_write}. Dữ liệu \texttt{rf\_write\_data} được đưa vào cổng ghi của tập thanh ghi cùng địa chỉ đích \texttt{rf\_rd\_addr = mem\_wb\_rd\_addr} để chốt ghi đồng bộ.
\end{itemize}

\section{Thiết kế Khối điều khiển (Control Unit) đơn nhân}
Khối điều khiển (\texttt{control\_unit.v}) chịu trách nhiệm giải mã mã lệnh 32-bit nhận được từ bộ nhớ chương trình, sinh ra các tín hiệu điều khiển tương ứng để cấu hình đường truyền dữ liệu (Datapath), ALU, khối giao tiếp bộ nhớ và bộ tạo hằng số mở rộng.

\subsection{Danh sách các tín hiệu điều khiển}
Các tín hiệu điều khiển chính được tạo ra bao gồm:
\begin{itemize}
    \item \textbf{\texttt{reg\_write}}: Cho phép ghi kết quả vào tập thanh ghi đa năng.
    \item \textbf{\texttt{alu\_src}}: Lựa chọn nguồn vào thứ hai (B) cho ALU ($0$: đọc từ thanh ghi nguồn \texttt{rs2\_data}, $1$: sử dụng hằng số mở rộng \texttt{imm}).
    \item \textbf{\texttt{alu\_control[3:0]}}: Xác định mã phép toán cho bộ ALU thực thi (ví dụ: cộng, trừ, dịch bit, toán tử logic).
    \item \textbf{\texttt{mem\_read}} / \textbf{\texttt{mem\_write}}: Tín hiệu yêu cầu đọc/ghi dữ liệu tương ứng đối với bộ nhớ RAM chia sẻ.
    \item \textbf{\texttt{mem\_to\_reg}}: Xác định nguồn ghi ngược trở lại thanh ghi đích \texttt{rd} ($0$: kết quả từ ALU, $1$: dữ liệu đọc lên từ bộ nhớ dữ liệu).
    \item \textbf{\texttt{branch}} / \textbf{\texttt{branch\_type[2:0]}}: Tín hiệu kích hoạt rẽ nhánh có điều kiện và loại điều kiện rẽ nhánh (BEQ, BNE, BLT, BGE).
    \item \textbf{\texttt{jump}} / \textbf{\texttt{jalr}}: Tín hiệu điều khiển nhảy không điều kiện (JAL) hoặc nhảy gián tiếp qua thanh ghi (JALR).
    \item \textbf{\texttt{lui}} / \textbf{\texttt{auipc}}: Các tín hiệu đặc biệt cho các chỉ thị nạp hằng số lớn vào nửa cao thanh ghi.
    \item \textbf{\texttt{mem\_lr}} / \textbf{\texttt{mem\_sc}} / \textbf{\texttt{mem\_amo}} / \textbf{\texttt{amo\_op[3:0]}}: Các tín hiệu kích hoạt và xác định toán tử nguyên tử tương ứng trong tập lệnh mở rộng nhóm A-Extension để thực hiện đồng bộ đa nhân.
\end{itemize}

\subsection{Bộ tạo hằng số mở rộng (Immediate Generator)}
Bộ tạo hằng số tích hợp bên trong khối điều khiển tự động trích xuất các trường bit của lệnh và thực hiện mở rộng dấu (sign-extension) thành số 32-bit hoàn chỉnh tùy theo định dạng của từng nhóm lệnh:
\begin{itemize}
    \item \textbf{I-type} (ADDI, LW, JALR): Trích xuất \texttt{instruction[31:20]}.
    \item \textbf{S-type} (SW): Trích xuất và ghép nối \texttt{\{instruction[31:25], instruction[11:7]\}}.
    \item \textbf{B-type} (BEQ...): Trích xuất, sắp xếp lại các trường bit và chèn thêm bit $0$ ở cuối để tính địa chỉ rẽ nhánh nửa từ.
    \item \textbf{U-type} (LUI, AUIPC): Trích xuất \texttt{instruction[31:12]} và nối thêm 12 bit $0$ vào nửa thấp.
    \item \textbf{J-type} (JAL): Trích xuất và sắp xếp lại các bit của khoảng dịch địa chỉ 20-bit, sau đó dịch trái 1 bit (chèn bit $0$ thấp).
\end{itemize}

\section{Cơ chế xử lý xung đột trong Pipeline}
\begin{itemize}
    \item \textbf{Data Hazard:} Giải quyết bằng cách chèn các mạch \textbf{Forwarding} (bơm tắt kết quả từ tầng EX/MEM và MEM/WB trực tiếp về đầu vào ALU tầng EX). Đối với xung đột \textbf{Load-use hazard}, bộ xử lý chèn 1 chu kỳ \textbf{Stall} (hoãn PC và xóa tín hiệu tầng ID/EX).
    \item \textbf{Control Hazard:} Khi phát hiện branch/jump đi sai hướng, toàn bộ các lệnh đang được nạp dở ở tầng IF/ID và ID/EX sẽ bị xóa (\textbf{Flush}) bằng cách đặt cờ hiệu valid của tầng đệm về 0, chuyển lệnh thành chỉ thị rỗng NOP.
\end{itemize}

\section{Thiết kế Bộ phân xử Arbiter và Bộ điều khiển Bus}
Trong hệ thống đa nhân chia sẻ bộ nhớ RAM dùng chung, bộ phân xử (Arbiter) và bộ điều khiển Bus (Bus Controller) đóng vai trò trung tâm để điều phối và quản lý việc truy cập bus bộ nhớ một cách hợp lệ, tránh xung đột dữ liệu.

\subsection{Thiết kế Bộ phân xử Round-Robin (\texttt{arbiter.v})}
Bộ phân xử sử dụng thuật toán xoay vòng Round-Robin để đảm bảo tính công bằng và tránh tình trạng đói bus (starvation) của các lõi. Nguyên lý hoạt động như sau:
\begin{itemize}
    \item Con trỏ ưu tiên \texttt{priority\_ptr} (3-bit) chỉ ra core có mức ưu tiên cao nhất trong chu kỳ hiện tại.
    \item Khi có nhiều core gửi yêu cầu truy cập cùng lúc (\texttt{request}), bộ phân xử sẽ quét xoay vòng từ vị trí con trỏ ưu tiên để tìm core đầu tiên có yêu cầu hợp lệ.
    \item Quyền truy cập (\texttt{grant}) chỉ được cấp cho core chiến thắng trong đúng 1 chu kỳ xung nhịp (pulse), ngăn chặn một lõi giữ bus quá lâu làm nghẽn hệ thống.
    \item Sau khi cấp quyền, con trỏ ưu tiên được cập nhật đến vị trí kế tiếp: \texttt{priority\_ptr <= winner + 3'd1}.
    \item Cơ chế \texttt{last\_granted} được thiết kế để yêu cầu mỗi core phải nhả request (\texttt{deassert}) trước khi có thể gửi một yêu cầu mới, giúp các core khác có cơ hội được phục vụ.
\end{itemize}

Dưới đây là mã nguồn Verilog hoàn chỉnh của khối Arbiter:

\begin{lstlisting}[caption={Mã nguồn Verilog bộ phân xử Round-Robin (\texttt{arbiter.v})}, label={lst:arbiter}]
module arbiter (
    input  wire        clk,
    input  wire        reset,
    input  wire [7:0]  request,
    output reg  [7:0]  grant,
    output reg         grant_valid,
    output reg  [2:0]  grant_id
);
    reg [2:0] priority_ptr;
    reg [7:0] last_granted;
    reg [2:0] winner;
    reg       found;
    integer   i;
    reg [2:0] candidate;

    // Tao masked request: chi xet request moi chua duoc grant
    wire [7:0] new_request = request & ~last_granted;

    always @(*) begin
        found  = 1'b0;
        winner = 3'd0;
        for (i = 0; i < 8; i = i + 1) begin
            candidate = priority_ptr + i[2:0];
            if (!found && new_request[candidate]) begin
                winner = candidate;
                found  = 1'b1;
            end
        end
    end

    always @(posedge clk or posedge reset) begin
        if (reset) begin
            grant        <= 8'b0;
            grant_valid  <= 1'b0;
            grant_id     <= 3'd0;
            priority_ptr <= 3'd0;
            last_granted <= 8'b0;
        end
        else begin
            if (found) begin
                grant        <= (8'b0000_0001 << winner);
                grant_valid  <= 1'b1;
                grant_id     <= winner;
                priority_ptr <= winner + 3'd1;
                last_granted <= (8'b0000_0001 << winner);
            end
            else begin
                grant        <= 8'b0;
                grant_valid  <= 1'b0;
                grant_id     <= 3'd0;
                last_granted <= last_granted & request;
            end
        end
    end
endmodule
\end{lstlisting}

\subsection{Thiết kế Bộ điều khiển Bus (\texttt{bus\_ctrl.v})}
Bộ điều khiển bus đóng vai trò như một bộ định tuyến dữ liệu (router) tổ hợp giữa 8 cổng của lõi và cổng duy nhất của bộ nhớ RAM chia sẻ. Dựa vào kết quả cấp quyền của Arbiter (\texttt{bus\_grant} và \texttt{bus\_grant\_id}):
\begin{itemize}
    \item Bộ điều khiển thực hiện ánh xạ (multiplex) các tín hiệu địa chỉ \texttt{mem\_addr}, dữ liệu ghi \texttt{mem\_wdata}, và các tín hiệu điều khiển ghi/đọc từ core được cấp quyền đến RAM dùng chung.
    \item Dữ liệu đọc từ RAM (\texttt{dm\_rdata}) được phản hồi ngược lại cho chính xác core đang nắm giữ quyền truy cập.
    \item Hỗ trợ giao dịch nguyên tử: tích hợp các bộ chốt reservation cho lệnh \texttt{LR.W} và \texttt{SC.W} để hỗ trợ các phép toán đồng bộ hóa phần mềm (như spinlock), tránh xung đột cập nhật dữ liệu.
\end{itemize}

\section{Tích hợp toàn hệ thống đa lõi 8-Core (\texttt{top\_8core.v})}
Khối tích hợp mức cao nhất (\texttt{top\_8core.v}) thực hiện liên kết 8 lõi CPU đơn nhân với các khối kết nối Interconnect và bộ nhớ dùng chung. Các lõi được sinh ra tự động thông qua khối lệnh \texttt{generate} trong Verilog để đảm bảo tính modular và dễ mở rộng.

Đoạn mã Verilog mô tả việc khởi tạo 8 lõi CPU trong \texttt{top\_8core.v}:

\begin{lstlisting}[caption={Khởi tạo 8 lõi CPU bằng khối lệnh generate trong \texttt{top\_8core.v}}, label={lst:top_8core}]
    genvar gi;
    generate
        for (gi = 0; gi < 8; gi = gi + 1) begin : gen_core
            risc_core #(
                .CORE_ID      (gi),
                .INIT_FILE    (PROGRAM_FILE),
                .PROGRAM_WORDS(PROGRAM_WORDS)
            ) u_core (
                .clk          (clk),
                .reset        (reset),
                .mem_req      (core_mem_req[gi]),
                .mem_addr     (core_addr[gi]),
                .mem_wdata    (core_wdata[gi]),
                .mem_we       (core_we[gi]),
                .mem_re       (core_re[gi]),
                .mem_lr       (core_lr[gi]),
                .mem_sc       (core_sc[gi]),
                .mem_amo      (core_amo[gi]),
                .mem_amo_op   (core_amo_op[gi]),
                .mem_rdata    (core_rdata[gi]),
                .mem_ready    (core_ready[gi]),
                .mem_sc_result(core_sc_result[gi])
            );
        end
    endgenerate
\end{lstlisting}

Các bus địa chỉ và dữ liệu giữa 8 lõi CPU được khai báo dưới dạng mảng (\texttt{wire [31:0] core\_addr [0:7]}) và được kết nối trực tiếp với bộ điều khiển bus \texttt{bus\_ctrl.v} và bộ phân xử \texttt{arbiter.v} để hoàn thành hệ thống đa nhân hoàn chỉnh.
