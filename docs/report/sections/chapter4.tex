\chapter{Hệ Thống 8-Core và Giao Tiếp Đa Lõi}
\label{chap:implementation_8core}

Chương này trình bày kiến trúc kết nối 8 lõi xử lý thành một hệ thống đa nhân hoàn chỉnh chia sẻ một bộ nhớ RAM dữ liệu dủng chung. Hệ thống sử dụng bộ phân xử (Arbiter) và bộ điều khiển (Bus Controller) để quản lý tranh chấp và hỗ trợ cơ chế đồng bộ đa luồng.

\section{Sơ đồ cấu trúc tổng thể}
Tại top-level (\texttt{top\_8core.v}), 8 lõi \texttt{risc\_core} được khởi tạo song song thông qua cấu trúc \texttt{generate}. Mỗi lõi sở hữu một bộ nhớ lệnh cục bộ (1 KB) để loại bỏ xung đột nạp lệnh, nhưng chia sẻ chung bộ nhớ dữ liệu (4 KB) thông qua trục liên kết (Interconnect).

\section{Bộ phân xử Round-Robin (\texttt{arbiter.v})}
\textbf{Mục đích:} Cấp quyền truy cập bus cho 8 lõi một cách công bằng, ngăn chặn tình trạng đói bus (starvation).
\textbf{Thuật toán:} Sử dụng con trỏ ưu tiên 3-bit xoay vòng. Khi có yêu cầu, arbiter tìm kiếm lõi có số thứ tự bằng hoặc lớn hơn con trỏ ưu tiên hiện tại. Chỉ cấp quyền (grant) đúng 1 chu kỳ để lõi đó bắt đầu giao dịch.

\begin{lstlisting}[caption={Thuật toán Round-Robin trong Arbiter},label={lst:arbiter}]
// Xoay vòng tìm lõi yêu cầu hợp lệ
for (i = 0; i < 8; i = i + 1) begin
    idx = (priority_ptr + i) % 8;
    if (request[idx]) begin
        grant_id = idx;
        grant_valid = 1'b1;
        break;
    end
end
// Cập nhật con trỏ sau khi cấp quyền
if (grant_valid) priority_ptr <= grant_id + 1;
\end{lstlisting}

\section{Bộ điều khiển Bus (\texttt{bus\_ctrl.v})}
\textbf{Mục đích:} Định tuyến địa chỉ, dữ liệu và điều khiển RAM dùng chung. Quản lý trạng thái giao dịch (FSM) bao gồm đọc/ghi thông thường, LR/SC và AMO.
\textbf{FSM:} Bus Controller duy trì máy trạng thái hữu hạn để xử lý tính nguyên tử.
\begin{itemize}
    \item \textbf{IDLE:} Chờ \texttt{grant} từ Arbiter.
    \item \textbf{ACTIVE:} Lõi được cấp quyền gửi dữ liệu. Với giao dịch thường, hoàn tất trong 1 chu kỳ. Với AMO, chuyển sang pha đọc.
    \item \textbf{AMO\_READ:} Gửi địa chỉ đọc lên RAM.
    \item \textbf{AMO\_WRITE:} Tính toán kết quả AMO (cộng, logic, max/min) và ghi lại RAM, hoàn tất giao dịch.
\end{itemize}

\subsection{Cơ chế Load-Reserved / Store-Conditional (LR/SC)}
Để hỗ trợ đồng bộ hóa, bộ điều khiển cấp phát 8 thanh ghi \texttt{reservation\_addr} và 8 cờ \texttt{reservation\_valid} cho 8 lõi.
\begin{itemize}
    \item Lệnh \textbf{LR} đặt cờ \texttt{valid=1} và ghi nhớ địa chỉ.
    \item Lệnh \textbf{SC} kiểm tra nếu \texttt{valid==1} và địa chỉ trùng khớp thì cho phép ghi, trả về 0 (thành công). Ngược lại không ghi, trả về 1 (thất bại).
    \item Mọi lệnh \textbf{Store} hoặc \textbf{AMO} hợp lệ chạm vào địa chỉ đang được reserve bởi một lõi khác sẽ xóa cờ \texttt{valid} của lõi đó.
\end{itemize}

\subsection{Cơ chế Atomic Memory Operations (AMO)}
AMOADD là lệnh chủ lực dùng để kiểm tra việc cộng dồn biến đếm an toàn. Hardware tự động khóa bus trong quá trình đọc-cộng-ghi.
\begin{lstlisting}[caption={Tính toán kết quả AMOADD trong FSM},label={lst:amoadd}]
if (amo_op == AMO_ADD) begin
    ram_wdata <= ram_rdata + active_wdata; // rdata (cũ) + rs2
    core_rdata <= ram_rdata; // Trả về giá trị cũ cho core
end
\end{lstlisting}
