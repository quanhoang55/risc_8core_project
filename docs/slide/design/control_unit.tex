    \begin{block}{Nguyên lý điều khiển luồng trên các Thanh ghi đệm Pipeline}
        \scriptsize
        \begin{itemize}
            \item \textbf{Clock Synchronization:} Dữ liệu/lệnh từ tầng trước chỉ được phép dịch chuyển sang tầng sau chính xác tại sườn lên của xung Clock (\texttt{posedge clk}).
            \item \textbf{Pipeline register:} Giữ cố định trạng thái xử lý độc lập của 5 lệnh khác nhau cùng lúc, ngăn chặn hiện tượng lấn át hay trộn lẫn dữ liệu giữa các tầng Datapath.
            \item \textbf{Pipeline Interlock:} Là nơi trực tiếp tiếp nhận các lệnh (\texttt{hold} để tạo \texttt{Stall}, hoặc \texttt{clear} để tạo \texttt{Flush/Bubble}) từ bộ điều khiển trung tâm.
        \end{itemize}
    \end{block}
\end{frame}

% --- TẦNG 1: IF/ID ---
\begin{frame}{PIPELINE REGISTERS: Fetch / Decode (IF/ID Register)}
    \begin{itemize}
        \item \textbf{Vai trò:} Chốt giữ \texttt{Instruction} 32-bit vừa đọc ra từ \texttt{Instruction Memory} và địa chỉ \texttt{PC} tương ứng để chuyển giao sang giai đoạn Decode.
        \item \textbf{Cơ chế Hazard:} Nhận tín hiệu \texttt{hold} khi kích hoạt \texttt{Load-Use Stall} hoặc \texttt{Memory Stall} để đóng băng trạng thái; nhận tín hiệu \texttt{clear} để thực hiện \texttt{Flush} lệnh rác khi rẽ nhánh \texttt{Taken} sai.
    \end{itemize}

\begin{frame}{PIPELINE REGISTERS Decode / Execute (ID/EX Register)}
    \begin{itemize}
        \item \textbf{Vai trò:} Tiếp nhận và chuyển tiếp toàn bộ các tín hiệu điều khiển từ \texttt{Control Unit}, giá trị các toán hạng đọc từ \texttt{Register File} (\texttt{rs1\_data}, \texttt{rs2\_data}) và số \texttt{Immediate} đã mở rộng sang giai đoạn \texttt{Execute} cho \texttt{ALU}.
        \item \textbf{Cơ chế Hazard:} Nhận tín hiệu \texttt{clear = 1} để chủ động chèn một \texttt{Bubble} (\texttt{NOP}) vào tầng \texttt{Execute} khi xảy ra xung đột \texttt{Load-Use}.
    \end{itemize}

\begin{frame}{PIPELINE REGISTERS Execute / Memory (EX/MEM Register)}
    \begin{itemize}
        \item \textbf{Vai trò:} Chốt giữ kết quả tính toán từ đầu ra của \texttt{ALU}, dữ liệu cần ghi vào \texttt{Data Memory} (\texttt{rs2\_data}) của lệnh \texttt{Store}, và địa chỉ thanh ghi đích \texttt{rd}.

    \end{itemize}

\begin{frame}{PIPELINE REGISTERS Memory / Writeback (MEM/WB Register)}
    \begin{itemize}
        \item \textbf{Vai trò:} Chốt giữ dữ liệu đọc lên từ \texttt{Data Memory} (\texttt{rdata}) hoặc kết quả trực tiếp từ \texttt{ALU} để sẵn sàng đưa về cổng ghi của \texttt{Register File}.
        \item \textbf{Cơ chế Hazard (Đã vá lỗi):} Loại bỏ hoàn toàn chân \texttt{clear} độc hại, chuyển sang sử dụng chân \texttt{hold} đồng bộ. Khi gặp \texttt{Memory Stall}, dữ liệu tầng này đứng im để bảo toàn dữ liệu, tránh lỗi mất thông tin trên \texttt{Pipeline}.
    \end{itemize}
